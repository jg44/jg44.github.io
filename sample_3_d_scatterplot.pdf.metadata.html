install.packages("scatterplot3d")
library(scatterplot3d)

# example points
x <- c(1,4,3,6,2,5)
y <- c(2,2,4,3,5,9)
z <- c(1,3,5,9,2,2)

# plot    
x11()
s <- scatterplot3d(x,y,z, type="p", pch=19, xlab = "Scale density",
              ylab="Defect density", zlab="Total lesion area")

# fit the model
ff = lm(z ~ x + y) ## in ff$coefficients are the line paramters z, mx, ny
s$plane3d(ff, col=2)
